    ```typescript
    "use client";

    import type { ConditionCompLink } from "@/lib/types";
    import { formatCurrency } from "@/lib/format";

    interface InlineCompsProps {
      compLinks: {
        original: ConditionCompLink;
        updated: ConditionCompLink;
        remodeled: ConditionCompLink;
      };
    }

    /**
     * Minimal inline display of 2-3 nearby recent sales.
     * Shows address + sold price in a compact list.
     * No external links -- everything stays on page (PAGE-04, COMP-04).
     * Deduplicates comps that share the same address.
     */
    export function InlineComps({ compLinks }: InlineCompsProps) {
      // Deduplicate -- if two condition tiers reference the same comp, show it once
      const seen = new Set<string>();
      const comps: { address: string; soldPrice: number }[] = [];

      for (const link of [compLinks.original, compLinks.updated, compLinks.remodeled]) {
        if (!seen.has(link.address) && link.soldPrice) {
          seen.add(link.address);
          comps.push({ address: link.address, soldPrice: link.soldPrice });
        }
      }

      if (comps.length === 0) return null;

      return (
        <div className="space-y-1.5 mb-4">
          <p className="text-xs text-charcoal/50 font-body uppercase tracking-wider">
            Recent sales near you
          </p>
          {comps.map((comp) => (
            <div key={comp.address} className="flex justify-between text-sm text-charcoal/70 font-body">
              <span className="truncate mr-4">{comp.address}</span>
              <span className="font-medium text-charcoal/80 whitespace-nowrap">
                {formatCurrency(comp.soldPrice)}
              </span>
            </div>
          ))}
        </div>
      );
    }
    ```

    Key design decisions:
    - Simple list, not cards (per research recommendation and Josh's "not busy" requirement)
    - `text-xs uppercase tracking-wider` for the "Recent sales near you" label -- small and unobtrusive
    - `text-sm text-charcoal/70` for comp entries -- secondary visual weight, doesn't compete with condition picker or value display
    - `truncate` on address for long addresses on mobile
    - Deduplication -- if the same comp serves multiple tiers, show it only once
    - Returns null if no comps have soldPrice (backward compat for old records)
    - NO external links anywhere (COMP-04, PAGE-04)
