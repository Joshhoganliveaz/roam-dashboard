    ```typescript
    export function getConditionValues(
      valueLow: number,
      valueHigh: number,
      baseValue?: number
    ): { original: number; updated: number; remodeled: number } {
      const base = baseValue ?? Math.round((valueLow + valueHigh) / 2);
      const LOWER_PCT = 0.03; // -3% for "similar to most"
      const UPPER_PCT = 0.05; // +5% for "best on the block"
      return {
        original: Math.round(base * (1 - LOWER_PCT)),
        updated: base,
        remodeled: Math.round(base * (1 + UPPER_PCT)),
      };
    }
    ```
    The asymmetry (3% down, 5% up) reflects Josh's SB7 insight that homeowners always think their home is worth more -- the upside feels slightly more generous. Using currentValue as the base (when provided) anchors around Josh's manual estimate rather than the comp midpoint.
