  - property1MortgageBalance = calcRemainingBalance(loanAmount, interestRate, loanTerm, ownershipMonths + year*12)
  - property2Value = property2Price * (1 + appreciationRate)^year
  - property2MortgageBalance = calcRemainingBalance(property2Mortgage, marketRate, 30, year*12)
  - rentalIncome = monthlyRent * (1 + rentGrowthRate)^year
  - helocBalance = helocDrawn (interest-only, principal unchanged)
  - monthlyHelocPayment = helocDrawn * (marketRate / 12)
  - property2MonthlyPayment = calcMortgagePayment(property2Mortgage, marketRate, 30)
  - netMonthlyCashFlow = rentalIncome - monthlyMortgage - monthlyHelocPayment - property2MonthlyPayment
  - totalEquity = (property1Value - property1MortgageBalance - helocBalance) + (property2Value - property2MortgageBalance)
- If helocDrawn <= 0, return empty array (no combo possible)
